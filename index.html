Options 

$(function() {
  $("#expand_accordion").expandAccordion({
    accordionHeight: 400,
    itemHeaderHeight: 44,
    headerHeight: 88,
    footerHeight: 180
  });
});

var defaults = {
  accordionHeight: 548,
  itemHeaderHeight: 44,
  headerHeight: undefined,
  footerHeight: undefined
};
